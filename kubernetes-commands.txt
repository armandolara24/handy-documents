*-*-* Networking in Kubernetes *-*-*

# macOS show network configuration 
ipconfig -a
# Linux show network configuration 
ip addr show

# within a k8s cluster there is a network different than the local computer
# a pod would have its own unique IP address, it doesn't matter where it's run
# it will always have a k8s-internal IP address.
# By defaultm each pod can access all other pods/services.
# This allows to think about each pod as a different physical machine.

# get more info about pods
kubectl get pods -o wide

kubectl exec -it wordpress-deployment-5c5c4c775b-cbbzs -- /bin/bash
apt update
apt install iputils-ping
# ping mariadb pod
ping 10.1.0.31

# get IP addresses from workloads
kubectl get endpoints

*-*-* Services *-*-*

# services are a mechanism to make pods accessible from within a kubernetes network.
# IP addresses will be the same even if pods change.

# create service for deployment
kubectl expose deployment wordpress-deployment
kubectl get services
apt install iputils-links
links http://[service-IP]

# DNS in Kubernetes
apt update && apt install -y dnsutils
nslookup mariadb-service

# Servces type 
ClusterIP (default): Cluster internal connections.
LoadBalancer: Incoming connections.
NodePort: ClusterIP + opens a port on all our nodes to public (30,000 -> 32,767)

kubectl delete service my-service
